# AE: Provides making Ankara the capital for Anatolian nations.
country_decisions = {
	make_ankara_capital = {
		potential = {
			any_province = {
				region = anatolia_region
			}
			OR = {
				NOT = {
					capital = 326
				}
				AND = {
					capital = 326
					NOT = {
						has_global_flag = ankara_founded
					}
				}
			}
			NOT = {
				has_country_flag = relocated_capital_ankara
			}
		}
		provinces_to_highlight = {
			region = anatolia_region
			NOT = {
				owned_by = ROOT
			}
		}
		allow = {
			anatolia_region = {
				owned_by = ROOT
			}
		}
		effect = {
			set_country_flag = relocated_capital_ankara
			add_stability = 1
			326 = {
				move_capital_effect = yes
				change_culture = ROOT
				change_religion = ROOT
			}
			if = {
				limit = {
					NOT = {
						has_global_flag = ankara_founded
					}
				}
				326 = {
					add_base_tax = 5
					add_base_production = 5
					add_base_manpower = 5
				}
				set_global_flag = ankara_founded
			}
			add_country_modifier = {
				name = "centralization_modifier"
				duration = 7300
			}
			add_prestige = 25
		}
		ai_will_do = {
			factor = 1
			modifier = {
				factor = 0
				development_in_provinces = {
					value = 200
					NOT = {
						region = anatolia_region
					}
				}
				capital = {
					NOT = {
						region = anatolia_region
					}
				}
			}
		}
	}
}
